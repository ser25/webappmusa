│   └── layout/
│       ├── Header.tsx
│       └── Navigation.tsx
├── hooks/
│   ├── useTelegram.ts
│   ├── useChat.ts
│   └── useMeditation.ts
├── services/
│   ├── telegram/
│   │   ├── TelegramService.ts
│   │   └── types.ts
│   ├── chat/

├── utils/
